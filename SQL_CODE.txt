CREATE TABLE CUSTOMER(
CUST_ID NUMBER(10) NOT NULL,
NAME VARCHAR2(30) NOT NULL,
PHONE NUMBER(11),
ADDRESS VARCHAR2(90),
DOB DATE,
CONSTRAINT CUSTOMER_PK PRIMARY KEY (CUST_ID)
);

CREATE TABLE COMPANY(
BRANCH_ID NUMBER(10) NOT NULL,
PHONE NUMBER(11),
ADDRESS VARCHAR2(90),
CONSTRAINT BRANCH_PK PRIMARY KEY (BRANCH_ID)
);

CREATE TABLE DRUGS(
NAME VARCHAR2(30) NOT NULL,
COST_PRICE FLOAT(11),
SELLING_PRICE FLOAT(11),
TOTAL_QUANTITY NUMBER(11),
MANUFACTURER VARCHAR2(30),
CONSTRAINT NAME_PK PRIMARY KEY (NAME)
);

CREATE TABLE BUYSFROM(BRANCH_ID NUMBER(10),CUST_ID NUMBER(10),
CONSTRAINT FK_BRANCH FOREIGN KEY (BRANCH_ID) REFERENCES COMPANY(BRANCH_ID),
CONSTRAINT FK_CUSTOMER FOREIGN KEY (CUST_ID) REFERENCES CUSTOMER(CUST_ID)
);

CREATE TABLE BUYS(
NAME VARCHAR2(30),
CUST_ID NUMBER(10),
DATE_OF_BUYING DATE,
QUANTITY_BOUGHT NUMBER(10),
CONSTRAINT FK_NAME FOREIGN KEY (NAME) REFERENCES DRUGS(NAME),
CONSTRAINT FK_CUSTOMER2 FOREIGN KEY (CUST_ID) REFERENCES CUSTOMER(CUST_ID)
);

CREATE TABLE PURCHASE(
BRANCH_ID NUMBER(10),
NAME VARCHAR2(30),
CONSTRAINT FK_BRANCH2 FOREIGN KEY (BRANCH_ID) REFERENCES COMPANY(BRANCH_ID),
CONSTRAINT FK_NAME2 FOREIGN KEY (NAME) REFERENCES DRUGS(NAME)
);

CREATE TABLE EXPIRY(
NAME VARCHAR2(30),
MANUFACTURING_DATE DATE,
EXPIRATION_DATE DATE,
CONSTRAINT FK_NAME3 FOREIGN KEY (NAME) REFERENCES DRUGS(NAME)
);

CREATE TABLE HISTORYSALES(
NAME VARCHAR2(30),
PRICE FLOAT(11),
QUANTITY_SOLD NUMBER(11),
CONSTRAINT FK_NAME4 FOREIGN KEY (NAME) REFERENCES DRUGS(NAME)
);
